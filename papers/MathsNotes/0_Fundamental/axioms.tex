\begin{definition}[associativity]\protect\hypertarget{def-associativity}{}\label{def-associativity}

For all \({a},{b},{c}\) in \(\Set\), one has
\(\Operation{(\Operation{a}{b})}{c}=\Operation{a}{(\Operation{b}{c})}\).

\end{definition}

\begin{definition}[identity]\protect\hypertarget{def-identity}{}\label{def-identity}

There exists an element \(\Identity\) in \(\Set\) such that, for every
\({a}\) in \(\Set\), one has \(\Operation{\Identity}{a}={a}\) and
\(\Operation{a}{\Identity}={a}\). Such an element is unique and is
called the \textbf{identity element}.

\end{definition}

\begin{definition}[inverse]\protect\hypertarget{def-Inverse}{}\label{def-Inverse}

For each \({a}\) in \(\Set\), there exists an element \({b}\) in
\(\Set\) such that \(\Operation{a}{b}=\Identity\) and
\(\Operation{b}{a}=\Identity\), where \(\Identity\) is the identity
element.

For each \({a}\), the element \({b}\) is unique and is called the
\textbf{inverse} of \({b}\) and is denoted \(\Inverse{a}\).

\end{definition}
